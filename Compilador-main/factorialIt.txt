program factorialIt;
var int : n, result;

function int factorial(int m)
var int : i;
{
    result = 1;
    i = 1;
    while (i <= m) do{
        result = result * i;
    }
    return(result);
}

main()
{
    write("input: ");
    read(n);
    result = factorial(n);
    write(result);
}
